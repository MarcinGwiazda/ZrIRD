DECLARE
    v_max_id   departments.department_id%TYPE;
    v_new_id   departments.department_id%TYPE;
    v_name     departments.department_name%TYPE := 'EDUCATION';
BEGIN
    SELECT MAX(department_id) INTO v_max_id FROM departments;

    v_new_id := v_max_id + 10;

    INSERT INTO departments (department_id,department_name,location_id)
    VALUES(v_new_id, v_name, 3000);
    DBMS_OUTPUT.PUT_LINE('Dodano departament o ID: ' || v_new_id);

END;

DECLARE
    CURSOR c_emp IS
        SELECT last_name, salary
        FROM employees
        WHERE department_id = 50;

    v_rec c_emp%ROWTYPE;
BEGIN
    OPEN c_emp;
    LOOP
        FETCH c_emp INTO v_rec;
        EXIT WHEN c_emp%NOTFOUND;

        IF v_rec.salary > 3100 THEN
            DBMS_OUTPUT.PUT_LINE(v_rec.last_name || ' - nie dawać podwyżki');
        ELSE
            DBMS_OUTPUT.PUT_LINE(v_rec.last_name || ' - dać podwyżkę');
        END IF;
    END LOOP;
    CLOSE c_emp;
END;
/

DECLARE
    CURSOR c_emp(p_min NUMBER, p_max NUMBER, p_name VARCHAR2) IS
        SELECT first_name, last_name, salary FROM employees
        WHERE salary BETWEEN p_min AND p_max
        AND LOWER(first_name) LIKE LOWER('%' || p_name || '%');

BEGIN
    DBMS_OUTPUT.PUT_LINE('--- 1000-5000, litera a ---');

    FOR rec IN c_emp(1000, 5000, 'a') LOOP
        DBMS_OUTPUT.PUT_LINE(rec.first_name || ' ' || rec.last_name);
    END LOOP;

    DBMS_OUTPUT.PUT_LINE('--- 5000-20000, litera u ---');

    FOR rec IN c_emp(5000, 20000, 'u') LOOP
        DBMS_OUTPUT.PUT_LINE(rec.first_name || ' ' || rec.last_name);
    END LOOP;
END;
/

CREATE OR REPLACE PROCEDURE p_add_job(
    p_id    jobs.job_id%TYPE,
    p_title jobs.job_title%TYPE
)
IS
BEGIN
    INSERT INTO jobs(job_id, job_title) VALUES (p_id, p_title);

    COMMIT;

EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Błąd: ' || SQLERRM);
END;
/

b.
CREATE OR REPLACE PROCEDURE p_aktualizuj_job(
    p_job_id    IN jobs.job_id%TYPE,
    p_nowy_tytul IN jobs.job_title%TYPE
)
IS
BEGIN
    UPDATE jobs SET job_title = p_nowy_tytul WHERE job_id = p_job_id;

    IF SQL%ROWCOUNT = 0 THEN
        RAISE_APPLICATION_ERROR(-20010, 'No jobs updated');
    END IF;

    DBMS_OUTPUT.PUT_LINE('Zaktualizowano stanowisko.');

END;
/

c.
CREATE OR REPLACE PROCEDURE p_delete_job(
    p_id jobs.job_id%TYPE)
IS
BEGIN
    DELETE FROM jobs WHERE job_id = p_id;

    IF SQL%ROWCOUNT = 0 THEN
        RAISE_APPLICATION_ERROR(-20011, 'No jobs deleted');
    END IF;

    COMMIT;
END;
/

CREATE OR REPLACE PROCEDURE p_get_emp(
    p_id      IN employees.employee_id%TYPE,
    p_salary  OUT employees.salary%TYPE,
    p_name    OUT employees.last_name%TYPE
)
IS
BEGIN
    SELECT salary, last_name INTO p_salary, p_name FROM employees WHERE employee_id = p_id;
END;
/

CREATE SEQUENCE emp_seq START WITH 1000;
CREATE OR REPLACE PROCEDURE p_add_employee(
    p_last_name employees.last_name%TYPE,
    p_salary    employees.salary%TYPE
)
IS
BEGIN
    IF p_salary > 20000 THEN
        RAISE_APPLICATION_ERROR(-20020, 'Wynagrodzenie za wysokie');
    END IF;

    INSERT INTO employees(employee_id, last_name, salary, hire_date, job_id)
    VALUES (emp_seq.NEXTVAL, p_last_name,p_salary, SYSDATE, 'IT_PROG');

END;
/
